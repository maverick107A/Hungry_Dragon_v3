// 
// Effect
//

technique Fog
{
    pass P0
    {
        //
        // Set Misc render states.

        pixelshader      = null;
        vertexshader     = null;
        fvf              = XYZ | Normal;
        Lighting         = false;
        NormalizeNormals = true;
        SpecularEnable   = false;

        //
        // Fog States

        FogVertexMode = LINEAR; // linear fog function
        FogStart      = 100.0f;  
        FogEnd        = 1000.0f;

        FogColor      = 0x00000000; // gray
        FogEnable     = true;       // enable
    }
}
